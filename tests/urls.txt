import java.util.HashMap;
import java.util.Map;

public class FindMatchingString {

    public static void main(String[] args) {
        // Create a HashMap with keys and corresponding strings
        Map<Integer, String> stringDict = new HashMap<>();
        stringDict.put(0, "dh2026pc01");
        stringDict.put(1, "dh2026pc02");
        stringDict.put(2, "dh2026pc03");
        stringDict.put(3, "dh2026pc04");

        // Specify the key for which you want to find the matching string
        int keyToFind = 2;

        // Check if the key exists in the map
        if (stringDict.containsKey(keyToFind)) {
            // Retrieve the matching string
            String matchingString = stringDict.get(keyToFind);
            System.out.println("Matching String for Key " + keyToFind + ": " + matchingString);
        } else {
            System.out.println("Key " + keyToFind + " not found in the dictionary.");
        }
    }
}
